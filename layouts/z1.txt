# Z1-Edit device descriptor — device identity + panel layout.
#
# Uncomplicated line-based format, not yet parsed by anything: this is a
# concrete transcription of real data (SysEx identity from defs.h, and the
# current hand-coded layout in z1Graphics.cpp's `dials[]` table), meant as
# the first real example to design the config parser/schema against.
#
# Syntax: space-separated key=value pairs per line (or bare key value for
# single-value lines like `device`/`manufacturerId`). Comments start with #,
# blank lines ignored. "gap N" inserts N points of extra space before the
# next dial (matches the one group-break the current code has, between the
# F1 and F2 dial groups).
#
# Protocol-wiring attributes on a dial (so mouseHandle.c never needs to know
# what a control *means*, only how to read/write it generically):
#   group=<n> param=<n>  — SysEx parameter group/ID, sent via a parameter-
#                           change message when the control has no cc=
#   cc=<n>                — MIDI CC number; when present, changes are sent as
#                           a real-time CC instead of a SysEx parameter change
#   nativeMax=<n>          — native/SysEx value range paired with a cc= dial
#                           (native = display * nativeMax / (max-1))
#   offset=<n>             — storage_value = display_value + offset (e.g. the
#                           "type" dials store 1-5 but display/drag as 0-4)
#   dumpOffset=<n>          — byte offset of this control within a full
#                           program-dump buffer (a different wire format from
#                           param= change messages); omit if not present there
#   dumpShift=<n>           — bits to shift right before masking (default 0)
#   dumpMask=<hex>          — mask applied after shifting (default 0xFF)
# `max` is always the count of display-space positions: values 0..max-1.
#
# "list <name> <items>" declares a device-wide named string list, not tied to
# any dial — e.g. patch-category names shown as plain text. Quote the item
# string if any item itself contains a space (comma still separates items).

# ── Device identity ──────────────────────────────────────────────────────────
# SysEx header: F0 42 (0x30|channel) 46 01 ... — see defs.h KORG_MANUFACTURER_ID /
# SYNTH_FAMILY_ID / SYNTH_MEMBER_ID / SYNTH_SYSEX_CHANNEL_BYTE.
device          Korg Z1
manufacturerId  0x42
familyId        0x46
memberId        0x01

# ── Named lists ───────────────────────────────────────────────────────────────
list category    Synth-Hard,Synth-Soft,E.Piano,Organ,Strings,Brass,Wind,Bell/Mallt,Guitar,Bass,Perc/Drums,Vocal,S.E./Natrl,Synth-Lead,Synth-Pad,Synth-Comp,Digital,User
list voiceMode    "Mono Multi,Mono Single,Poly"
list unisonType   "2 voices,3 voices,6 voices"

# ── Filters row (Synthesis page) ─────────────────────────────────────────────
page       synthesis
section    filters
dialSize   40
spacing    50

color route 0.8 0.8 0.3
color f1    0.2 0.6 1.0
color f2    0.3 0.9 0.5

dial route  label="Route"    color=route  max=3    names=SERI1,SERI2,PARA          group=1 param=258 dumpOffset=311 dumpMask=0x03
dial f2link label="F2 Link"  color=route  max=2    names=OFF,ON                    group=1 param=259 dumpOffset=311 dumpShift=2 dumpMask=0x01
dial f1type label="F1 Type"  color=f1     max=5    names=LPF,HPF,BPF,BRF,2BPF      group=1 param=261 offset=1 dumpOffset=312
dial f1trim label="F1 Trim"  color=f1     max=100  display=raw                     group=1 param=262 dumpOffset=313
dial f1cut  label="F1 Cut"   color=f1     max=128  display=ccnative                group=1 param=263 cc=0x55 nativeMax=99 dumpOffset=314
dial f1res  label="F1 Res"   color=f1     max=128  display=ccnative                group=1 param=274 cc=0x56 nativeMax=99 dumpOffset=325

gap 20

dial f2type label="F2 Type"  color=f2     max=5    names=LPF,HPF,BPF,BRF,2BPF      group=1 param=288 offset=1 dumpOffset=339
dial f2trim label="F2 Trim"  color=f2     max=100  display=raw                     group=1 param=289 dumpOffset=340
dial f2cut  label="F2 Cut"   color=f2     max=128  display=ccnative                group=1 param=290 cc=0x58 nativeMax=99 dumpOffset=341
dial f2res  label="F2 Res"   color=f2     max=128  display=ccnative                group=1 param=301 cc=0x59 nativeMax=99 dumpOffset=352
